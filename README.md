# IntroLib

# A Roblox Intro UI Library

# Features:

- Fade animation
- Icon animation
- Blur
- Cinematic bars
- Easy setup

# Install:

local Intro = loadstring(game:HttpGet("https://raw.githubusercontent.com/norixu/Introlib/refs/heads/main/IntroLib.lua"))()

Intro:Play({
Logo = "rbxassetid://YOUR_ICON",
Title = "Loading Example..."
})
